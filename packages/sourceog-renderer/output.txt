// sourceog-renderer/src/render.tsx
// Alibaba CTO 2027 Standard — Main Render Pipeline (Server Entry)

import React from 'react';
import type { RouteDefinition } from '@sourceog/router';
import type { SourceOGRequestContext, SourceOGResponse } from '@sourceog/runtime';
import { renderToPipeableStream } from 'react-dom/server';
import { PassThrough } from 'node:stream';
import { logger } from './core/logger.js';
import { renderer } from './orchestrator/renderer-orchestrator.js';
import { resolveManifestPathForRouteFile } from './manifests/manifest-resolver.js';

/**
 * Main server-side rendering function.
 * Handles the entire lifecycle of a route render request:
 * 1. Planning (Strategy/Policy)
 * 2. Orchestration (Worker Pool)
 * 3. RSC Flight Generation
 * 4. HTML Shell Generation (if applicable)
 */
export async function renderRoute(
  route: RouteDefinition,
  context: SourceOGRequestContext
): Promise<SourceOGResponse> {
  logger.debug(`Rendering route: ${route.id}`);

  // 1. Resolve Manifest
  const manifestPath = resolveManifestPathForRouteFile(route.file);

  // 2. Render RSC Payload via Worker Pool
  // This offloads the heavy React Server Component work to a worker thread.
  const rscResult = await renderer.renderRoute(route, context, {
    collectChunks: true, // We need the chunks for HTML generation
  });

  // 3. Generate HTML Shell (SSR)
  // We wrap the RSC payload in a basic HTML structure for the initial navigation.
  // In a full implementation, we would decode the flight stream to determine 
  // the head tags and hydration script.
  
  const stream = new PassThrough();
  
  // Write basic HTML preamble
  stream.write('<!DOCTYPE html><html><head><title>SourceOG</title></head><body><div id="root">');

  // Inject RSC Payload (script tags)
  for (const chunk of rscResult.chunks) {
    // Escape chunk for embedding in script tag
    const escaped = chunk.replace(/<\//g, '<\\/');
    stream.write(`<script type="text/x-component">${escaped}</script>`);
  }

  // Write closing tags
  stream.write('</div></body></html>');
  stream.end();

  return new SourceOGResponse(stream, {
    headers: {
      'Content-Type': 'text/html; charset=utf-8',
      'X-SourceOG-Render': 'rsc-shell',
    },
  });
}

// Re-export types and functions for public usage
export { renderer };

// sourceog-renderer/src/rsc-worker-bootstrap.mjs
// Alibaba CTO 2027 Standard — Worker Bootstrap (ESM Entry)

import { workerData, parentPort } from 'node:worker_threads';
import { pathToFileURL, fileURLToPath } from 'node:url';
import path from 'node:path';
import { createRequire } from 'node:module';
import { readFileSync, existsSync, mkdirSync, writeFileSync } from 'node:fs';
import { createHash } from 'node:crypto';
import { tmpdir } from 'node:os';

// ---------------------------------------------------------------------------
// Configuration
// ---------------------------------------------------------------------------

const USE_INLINE_TRANSFORM = workerData?.useInlineTransform === true;
const TRANSFORM_TMP_DIR = path.join(tmpdir(), 'sourceog-inline-transform');

if (!existsSync(TRANSFORM_TMP_DIR)) {
  try { mkdirSync(TRANSFORM_TMP_DIR, { recursive: true }); } catch {}
}

// ---------------------------------------------------------------------------
// Minimal Inline Transpiler (Fallback)
// ---------------------------------------------------------------------------

let esbuildTransform = null;
let sucraseTransform = null;

async function ensureTransformers() {
  if (esbuildTransform || sucraseTransform) return;
  
  // Try esbuild
  try {
    const esbuild = await import('esbuild');
    esbuildTransform = async (code, filename) => {
      const result = await esbuild.transform(code, {
        loader: filename.endsWith('.tsx') ? 'tsx' : filename.endsWith('.ts') ? 'ts' : 'js',
        target: 'es2022', format: 'esm', jsx: 'automatic',
      });
      return result.code;
    };
  } catch {}

  // Try sucrase
  try {
    const sucrase = await import('sucrase');
    sucraseTransform = async (code, filename) => {
      return sucrase.transform(code, { transforms: ['typescript', 'jsx'], jsxRuntime: 'automatic' }).code;
    };
  } catch {}
}

async function loadWithTransform(url) {
  await ensureTransformers();
  const fsPath = fileURLToPath(url);
  const source = readFileSync(fsPath, 'utf8');
  const hash = createHash('sha256').update(source).digest('hex').slice(0, 8);
  const tmpFile = path.join(TRANSFORM_TMP_DIR, `${path.basename(fsPath)}-${hash}.mjs`);
  
  if (!existsSync(tmpFile)) {
    const transform = esbuildTransform || sucraseTransform;
    if (!transform) throw new Error('No inline transformer found for worker bootstrap');
    const code = await transform(source, fsPath);
    writeFileSync(tmpFile, code, 'utf8');
  }
  
  return import(pathToFileURL(tmpFile).href);
}

// ---------------------------------------------------------------------------
// Entry Logic
// ---------------------------------------------------------------------------

async function main() {
  // The main entry is the compiled worker-entry.js
  // We calculate its path relative to this bootstrap file.
  
  // In dev, this might be .ts. In prod, .js.
  // We prioritize .js for production speed.
  
  const bootstrapDir = path.dirname(fileURLToPath(import.meta.url));
  
  // Try compiled JS first
  const jsEntry = path.join(bootstrapDir, 'workers', 'worker-entry.js');
  
  if (existsSync(jsEntry)) {
    return import(pathToFileURL(jsEntry).href);
  }

  // Try TS source (development with inline transform)
  const tsEntry = path.join(bootstrapDir, 'workers', 'worker-entry.ts');
  
  if (existsSync(tsEntry)) {
    if (USE_INLINE_TRANSFORM) {
      return loadWithTransform(pathToFileURL(tsEntry).href);
    } else {
      // If no inline transform, but file is TS, it relies on Node flags (--import tsx)
      // which should have been passed by the pool.
      // If we reach here without flags, it will likely fail, but we try importing.
      return import(pathToFileURL(tsEntry).href);
    }
  }

  throw new Error('Worker entry point not found.');
}

main().catch(err => {
  console.error('[SOURCEOG Bootstrap] Critical Failure:', err);
  process.exit(1);
});

// Re-export types used by legacy consumers

/**
 * @deprecated Direct usage is discouraged. Use the modular exports instead.
 */
export async function renderFlightStream(
  requestId: string,
  payload: WorkerRenderRequest,
  manifest: NormalizedClientManifest,
  onChunk?: (chunk: string) => void
): Promise<WorkerRenderResponse> {
  const { route, context, timeoutMs } = payload;
  
  try {
    // 1. Resolve Components
    const components = await resolveRouteComponents(route);
    
    // 2. Render
    const React = await import('react');
    const element = React.createElement(components.component as React.ComponentType, {
      params: context.params,
      searchParams: Object.fromEntries(context.query || []),
    });

    const chunks = await renderFlight(element, manifest, {
      routeId: route.id,
      timeoutMs: timeoutMs || 5000,
    });

    // Trigger onChunk callback for streaming if provided
    if (onChunk) {
      chunks.forEach(onChunk);
    }

    return {
      format: 'react-flight-text',
      chunks,
      usedClientRefs: [],
    };
  } catch (error) {
    throw toError(error);
  }
}

  if (!manifestPath || !existsSync(manifestPath)) {
    return {};

  try {
    const raw = readFileSync(manifestPath, 'utf8');
    const parsed = JSON.parse(raw);
    return parsed && typeof parsed === 'object' ? parsed : {};
  } catch {
    return {};
  }
}

/**
 * Helper to safely convert an unknown thrown value into a valid Error object.
 * Used to normalize exceptions caught in try/catch blocks.
 */
export function toError(value: unknown): Error {
  if (value instanceof Error) return value;
  
  // Handle non-Error throws (strings, objects, etc.)
  try {
    return new Error(String(value));
  } catch {
    // Fallback for complex objects
    return new Error('An unknown error occurred.');
  }
}

/**
 * Type guard to check if an object is a SourceOG error.
 */
export function isSourceOGError(error: unknown): error is SourceOGBaseError {
  return error instanceof SourceOGBaseError || 
         (typeof error === 'object' && error !== null && (error as any).isSourceOGError === true);
}

// Alibaba CTO 2027 Standard — Structured Logging

import { isProduction, isDebug } from './env.js';
import type { SourceOGBaseError } from '@sourceog/genbook/errors';

type LogLevel = 'debug' | 'info' | 'warn' | 'error';

const LOG_LEVEL_PRIORITY: Record<LogLevel, number> = {
  debug: 0,
  info: 1,
  warn: 2,
  error: 3,
};

const CURRENT_LOG_LEVEL: LogLevel = 
  isProduction ? 'warn' : 
  isDebug ? 'debug' : 'info';

/**
 * Formats a log entry for production (JSON) or development (Human Readable).
 */
function formatEntry(level: LogLevel, message: string, meta?: Record<string, unknown>): string {
  const timestamp = new Date().toISOString();
  
  if (isProduction) {
    return JSON.stringify({
      time: timestamp,
      level,
      msg: message,
      ...meta,
    });
  }

  // Development: Pretty print
  const prefix = `[SOURCEOG] [${level.toUpperCase()}]`;
  const metaStr = meta ? `\n${JSON.stringify(meta, null, 2)}` : '';
  return `${prefix} ${timestamp}: ${message}${metaStr}`;
}

/**
 * Extracts safe metadata from an error object.
 */
function extractErrorMeta(error: unknown): Record<string, unknown> {
  if (typeof error === 'object' && error !== null) {
    const isSGErr = (error as SourceOGBaseError).isSourceOGError === true;
    if (isSGErr) {
      return (error as SourceOGBaseError).toJSON();
    }
    // Standard Error
    const err = error as Error;
    return {
      name: err.name,
      message: err.message,
      stack: err.stack,
    };
  }
  return { error: String(error) };
}

export const logger = {
  debug(message: string, meta?: Record<string, unknown>): void {
    if (LOG_LEVEL_PRIORITY.debug < LOG_LEVEL_PRIORITY[CURRENT_LOG_LEVEL]) return;
    console.debug(formatEntry('debug', message, meta));
  },

  info(message: string, meta?: Record<string, unknown>): void {
    if (LOG_LEVEL_PRIORITY.info < LOG_LEVEL_PRIORITY[CURRENT_LOG_LEVEL]) return;
    console.info(formatEntry('info', message, meta));
  },

  warn(message: string, meta?: Record<string, unknown>): void {
    if (LOG_LEVEL_PRIORITY.warn < LOG_LEVEL_PRIORITY[CURRENT_LOG_LEVEL]) return;
    console.warn(formatEntry('warn', message, meta));
  },

  error(message: string, error?: unknown, meta?: Record<string, unknown>): void {
    const errorMeta = error ? extractErrorMeta(error) : {};
    const combinedMeta = { ...meta, ...errorMeta };
    console.error(formatEntry('error', message, combinedMeta));
  },
};

import path from 'node:path';

/**
 * Normalizes a file path to a `file://` URL.
 * Safe for Windows and Unix paths.
 * 
 * @param filepath - The absolute or relative path.
 * @returns A `file://` URL string.
 */
export function toFileUrl(filepath: string): string {
  if (filepath.startsWith('file://')) return filepath;
  return pathToFileURL(path.resolve(filepath)).href;
}

/**
 * Converts a `file://` URL back to an absolute filesystem path.
 * 
 * @param url - The `file://` URL string.
 * @returns An absolute filesystem path.
 */
export function toFsPath(url: string): string {
  if (!url.startsWith('file://')) return path.resolve(url);
  return fileURLToPath(url);
}

/**
 * Checks if a string is a valid file URL.
 */
export function isFileUrl(value: string): boolean {
  return typeof value === 'string' && value.startsWith('file://');
}

/**
 * Normalizes a manifest path to a URL suitable for worker communication.
 * Handles absolute paths and ensures `file://` scheme.
 * 
 * @param manifestPath - The path to the manifest file.
 * @returns A normalized URL string.
 */
export function toWorkerManifestUrl(manifestPath?: string): string {
  if (!manifestPath) return '';
  return toFileUrl(manifestPath);
}

/**
 * Resolves a relative import specifier against a parent path/URL.
 * 
 * @param specifier - The import specifier (relative path or URL).
 * @param parentPath - The parent file path or URL.
 * @returns A resolved `file://` URL.
 */
export function resolveImportUrl(specifier: string, parentPath: string): string {
  if (specifier.startsWith('file://') || specifier.startsWith('node:')) {
    return specifier;
  }

  const parentUrl = isFileUrl(parentPath) ? parentPath : toFileUrl(parentPath);
  const parentDir = path.dirname(toFsPath(parentUrl));
  
  if (path.isAbsolute(specifier)) {
    return toFileUrl(specifier);
  }
  
  // Handle relative paths
  return toFileUrl(path.resolve(parentDir, specifier));
}

// sourceog-renderer/src/manifests/client-reference-manifest.ts
// Alibaba CTO 2027 Standard — Client Manifest Normalization

import type { ClientReferenceManifestInput, NormalizedClientManifest, ClientManifestEntry } from '@sourceog/genbook/types';

/**
 * Normalizes the raw manifest input into a consistent key-value record.
 * 
 * Input formats supported:
 * 1. `{ "registry": { ... } }` wrapper.
 * 2. Flat key-value map.
 * 3. Path-keyed entries where the entry itself contains `id` and `name`.
 * 
 * Output keys:
 * - `${id}#${name}` (Composite key for exact lookup)
 * - `${id}` (Module root lookup)
 */
export function normalizeClientManifest(input: ClientReferenceManifestInput): NormalizedClientManifest {
  // Extract registry if wrapped
  const registry = 
    (input && typeof input === 'object' && 'registry' in input && input.registry) 
      ? input.registry 
      : input;

  const normalized: NormalizedClientManifest = {};

  for (const [sourceKey, rawEntry] of Object.entries(registry ?? {})) {
    if (!rawEntry || typeof rawEntry !== 'object') continue;

    const entry = rawEntry as Partial<ClientManifestEntry>;
    
    // Derive ID and Name
    const id = entry.id;
    const name = entry.name ?? (sourceKey.includes('#') ? sourceKey.split('#')[1] : 'default');

    if (!id) continue;

    const normalizedEntry: ClientManifestEntry = {
      id,
      name,
      chunks: entry.chunks ?? [],
      async: entry.async ?? false,
      filepath: entry.filepath,
    };

    // 1. Composite Key (Standard lookup)
    const compositeKey = `${id}#${name}`;
    normalized[compositeKey] = normalizedEntry;

    // 2. Module Root Key (Fallback lookup)
    // Only set if not already present to avoid overwriting specific exports
    if (!(id in normalized)) {
      normalized[id] = normalizedEntry;
    }
  }

  return normalized;
}

// sourceog-renderer/src/manifests/manifest-cache.ts
// Alibaba CTO 2027 Standard — Manifest Content Cache

import type { NormalizedClientManifest } from '@sourceog/genbook/types';
import { logger } from '../core/logger.js';
import { MANIFEST_CACHE_MAX } from '../core/constants.js';

/**
 * Entry for the manifest content cache.
 */
interface ManifestCacheEntry {
  manifest: NormalizedClientManifest;
  lastModified: number; // Used for validation
}

/**
 * LRU Cache for loaded manifest content.
 * 
 * Rationale:
 * Manifest files (JSON) can be large. Loading and parsing them from disk 
 * on every render request is inefficient. We cache the parsed object in memory.
 * 
 * This is separate from the Path Cache (in manifest-resolver) which only stores file paths.
 */
class ManifestContentCache {
  private cache = new Map<string, ManifestCacheEntry>();
  private max: number;

  constructor(max: number) {
    this.max = max;
  }

  get(key: string): NormalizedClientManifest | undefined {
    const entry = this.cache.get(key);
    if (!entry) return undefined;
    
    // LRU refresh
    this.cache.delete(key);
    this.cache.set(key, entry);
    return entry.manifest;
  }

  set(key: string, manifest: NormalizedClientManifest): void {
    if (this.cache.has(key)) this.cache.delete(key);
    
    this.cache.set(key, {
      manifest,
      lastModified: Date.now(),
    });

    if (this.cache.size > this.max) {
      const oldest = this.cache.keys().next().value!;
      this.cache.delete(oldest);
      logger.debug(`Evicted manifest from cache: ${oldest}`);
    }
  }

  has(key: string): boolean {
    return this.cache.has(key);
  }

  delete(key: string): boolean {
    return this.cache.delete(key);
  }

  clear(): void {
    this.cache.clear();
  }
}

// Singleton instance
export const manifestContentCache = new ManifestContentCache(MANIFEST_CACHE_MAX);

/**
 * Helper to generate a cache key.
 * We use the absolute file path as the key.
 */
export function getManifestCacheKey(filePath: string): string {
  return filePath; // Path should already be resolved and normalized

// sourceog-renderer/src/manifests/manifest-resolver.ts
// Alibaba CTO 2027 Standard — Manifest Path Resolution

import path from 'node:path';
import { existsSync } from 'node:fs';
import { ManifestTraversalError } from '@sourceog/genbook/errors';
import { PROJECT_ROOT, MANIFEST_CACHE_MAX } from '../core/constants.js';
import { logger } from '../core/logger.js';

/**
 * Simple LRU Cache implementation for manifest paths.
 * Bounded to prevent unbounded memory growth in projects with thousands of routes.
 */
class ManifestPathCache {
  private cache = new Map<string, string | null>();
  private max: number;

  constructor(max: number) {
    this.max = max;
  }

  get(key: string): string | null | undefined {
    if (!this.cache.has(key)) return undefined;
    const value = this.cache.get(key);
    // Re-insert to mark as recently used (LRU logic)
    this.cache.delete(key);
    if (value !== null && value !== undefined) {
      this.cache.set(key, value);
    }
    return value;
  }

  set(key: string, value: string | null): void {
    if (this.cache.has(key)) this.cache.delete(key);
    this.cache.set(key, value);
    if (this.cache.size > this.max) {
      // Evict oldest (first item in Map iterator)
      const oldest = this.cache.keys().next().value;
      if (oldest) this.cache.delete(oldest);
    }
  }
}

const manifestPathCache = new ManifestPathCache(MANIFEST_CACHE_MAX);

/**
 * Resolves the client-reference manifest path for a specific route file.
 * 
 * Search Strategy:
 * 1. Check cache.
 * 2. Walk up directory tree from route file to project root.
 * 3. Look for `.sourceog/manifests/client-reference-manifest.json`.
 * 4. Fallback to project root manifests.
 * 
 * Security:
 * - Enforces strict boundary checks to prevent path traversal outside PROJECT_ROOT.
 * 
 * @param routeFile - The absolute path to the route file.
 * @returns The resolved absolute path to the manifest, or undefined if not found.
 */
export function resolveManifestPathForRouteFile(routeFile: string): string | undefined {
  // 1. Check Cache
  const cached = manifestPathCache.get(routeFile);
  if (cached !== undefined) return cached ?? undefined;

  const normalizedRoot = PROJECT_ROOT + path.sep;
  const resolvedRouteFile = path.resolve(routeFile);

  // 2. Security Boundary Check
  if (
    path.isAbsolute(routeFile) &&
    !resolvedRouteFile.startsWith(normalizedRoot) &&
    resolvedRouteFile !== PROJECT_ROOT
  ) {
    throw new ManifestTraversalError(routeFile, PROJECT_ROOT);
  }

  // 3. Walk Up Directory Tree
  let currentDir = path.dirname(resolvedRouteFile);
  let previousDir = "";

  while (
    currentDir &&
    currentDir !== previousDir &&
    (currentDir.startsWith(normalizedRoot) || currentDir === PROJECT_ROOT)
  ) {
    const candidates = [
      path.join(currentDir, ".sourceog", "manifests", "client-reference-manifest.json"),
      path.join(currentDir, ".sourceog", "client-reference-manifest.json"),
    ];

    for (const candidate of candidates) {
      if (existsSync(candidate)) {
        logger.debug(`Manifest resolved for ${routeFile} at ${candidate}`);
        manifestPathCache.set(routeFile, candidate);
        return candidate;
      }
    }

    previousDir = currentDir;
    currentDir = path.dirname(currentDir);
  }

  // 4. Fallback to Project Root
  const rootCandidates = [
    path.join(PROJECT_ROOT, ".sourceog", "manifests", "client-reference-manifest.json"),
    path.join(PROJECT_ROOT, ".sourceog", "client-reference-manifest.json"),
  ];

  for (const candidate of rootCandidates) {
    if (existsSync(candidate)) {
      manifestPathCache.set(routeFile, candidate);
      return candidate;
    }
  }

  // Not found
  manifestPathCache.set(routeFile, null);
  return undefined;
}

/**
 * Clears the manifest path cache.
 * Useful for hot-reloading or testing.
 */
export function clearManifestCache(): void {
  // Recreate cache instance (simplest way to clear)
  (manifestPathCache as any).cache.clear();
}

// sourceog-renderer/src/orchestrator/admission-control.ts
// Alibaba CTO 2027 Standard — Admission Control & Load Shedding

import { WorkerPoolExhaustedError } from '@sourceog/genbook/errors';
import type { Priority } from './priority-queue.js';

/**
 * The result of an admission control check.
 */
export interface AdmissionDecision {
  /** If true, the request is accepted into the queue. */
  accepted: boolean;
  
  /** If false, contains the error to be thrown/rejected. */
  error?: Error;
  
  /** Optional adjusted priority for the request. */
  priority?: Priority;
}

/**
 * Configuration for the admission controller.
 */
export interface AdmissionControlConfig {
  /** Maximum depth of the queue before rejecting requests. */
  maxQueueDepth: number;
  
  /** Maximum concurrent workers. */
  maxWorkers: number;
  
  /** Threshold (0.0 - 1.0) at which to start shedding low-priority requests. */
  loadShedThreshold?: number;
}

/**
 * Determines if a request should be accepted based on current system load.
 * 
 * This protects the system from becoming overwhelmed (cascade failure) by 
 * failing fast when saturated.
 */
export class AdmissionController {
  private readonly config: AdmissionControlConfig;

  constructor(config: AdmissionControlConfig) {
    this.config = config;
  }

  /**
   * Evaluates a request against the current load.
   * 
   * @param currentQueueDepth - The current number of items waiting.
   * @param currentWorkerUsage - The number of active workers.
   * @param priority - The priority of the incoming request.
   */
  public check(
    currentQueueDepth: number,
    currentWorkerUsage: number,
    priority: Priority
  ): AdmissionDecision {
    const { maxQueueDepth, maxWorkers, loadShedThreshold = 0.8 } = this.config;
    
    // 1. Total Saturation Check
    // If queue is full, reject everything (Tail Drop).
    if (currentQueueDepth >= maxQueueDepth) {
      return {
        accepted: false,
        error: new WorkerPoolExhaustedError(currentQueueDepth, maxWorkers),
      };
    }

    // 2. Load Shedding (Smart Saturation)
    // If we are approaching capacity, reject low priority requests early.
    const activeHandles = currentWorkerUsage + currentQueueDepth;
    const capacity = maxWorkers + maxQueueDepth;
    const utilization = activeHandles / capacity;

    if (utilization > loadShedThreshold) {
      // Shed "Low" and "Deferred" requests if we are above threshold
      if (priority >= 3) { // 3 = Low, 4 = Deferred
        return {
          accepted: false,
          error: new WorkerPoolExhaustedError(currentQueueDepth, maxWorkers),
        };
      }
    }

    // 3. Accept
    return { accepted: true };
  }
}

// sourceog-renderer/src/orchestrator/pool-registry.ts
// Alibaba CTO 2027 Standard — Shared Pool Registry & Lifecycle

import type { WorkerPool, WorkerPoolOptions } from './worker-pool.js';
import { resolveManifestPathForRouteFile } from '../manifests/manifest-resolver.js';
import { POOL_TTL_MS, SWEEP_INTERVAL_MS } from '../core/constants.js';
import { logger } from '../core/logger.js';

/**
 * Internal entry for tracking pool usage and lifecycle.
 */
interface PoolEntry {
  pool: WorkerPool;
  lastUsedAt: number;
}

/**
 * Global registry of active worker pools, keyed by manifest path.
 * 
 * Rationale:
 * Different routes may require different manifest contexts (e.g., different apps in a monorepo).
 * This registry ensures we reuse pools with identical configurations while isolating distinct ones.
 */
const sharedWorkerPools = new Map<string, PoolEntry>();

/**
 * Timer reference for the stale pool sweeper.
 */
let sweepTimer: NodeJS.Timeout | undefined;

/**
 * Flag to ensure the process shutdown hook is installed exactly once.
 */
let shutdownHookInstalled = false;

/**
 * Retrieves a shared WorkerPool instance for the given route file.
 * 
 * If a pool for the route's manifest already exists, it is reused.
 * Otherwise, a new pool is created and registered.
 * 
 * @param routeFile - The absolute path to the route file.
 * @param options - Optional overrides for pool configuration.
 */
export function getSharedWorkerPool(
  routeFile: string,
  options?: WorkerPoolOptions
): WorkerPool {
  ensureSweepTimer();
  ensureShutdownHook();

  const now = Date.now();
  
  // Resolve manifest path for this route
  const manifestPath = resolveManifestPathForRouteFile(routeFile) ?? '__sourceog-default__';

  // Check for existing pool
  const existing = sharedWorkerPools.get(manifestPath);
  if (existing) {
    existing.lastUsedAt = now;
    return existing.pool;
  }

  // Create new pool
  logger.debug(`Creating new shared worker pool for manifest: ${manifestPath}`);
  
  // Dynamic import to avoid circular dependency if WorkerPool imports this registry
  // (In this file structure, it doesn't, but good practice for singletons)
  const { WorkerPool } = require('./worker-pool.js'); // Using require for sync loading if needed, or async init
  
  // Note: WorkerPool initialization is async, but we return instance immediately.
  // The caller must call `initialize()` or rely on the pool's internal lazy init.
  const pool = new WorkerPool({
    ...options,
    manifestPath: manifestPath === '__sourceog-default__' ? undefined : manifestPath,
  });

  sharedWorkerPools.set(manifestPath, { pool, lastUsedAt: now });
  
  // Trigger initialization (async) - pool handles mutex internally
  void pool.initialize();

  return pool;
}

/**
 * Starts the timer that periodically sweeps stale pools.
 * Stale pools are those that haven't been used for `POOL_TTL_MS`.
 */
function ensureSweepTimer(): void {
  if (sweepTimer) return;

  sweepTimer = setInterval(() => {
    const now = Date.now();
    for (const [key, entry] of sharedWorkerPools) {
      if (now - entry.lastUsedAt > POOL_TTL_MS) {
        logger.info(`Sweeping stale worker pool: ${key}`);
        sharedWorkerPools.delete(key);
        void entry.pool.shutdown();
      }
    }
  }, SWEEP_INTERVAL_MS);

  // Unref the timer so it doesn't prevent the process from exiting
  sweepTimer.unref();
}

/**
 * Installs the global process shutdown hook for graceful termination.
 */
function ensureShutdownHook(): void {
  if (shutdownHookInstalled) return;
  shutdownHookInstalled = true;

  process.once('SIGTERM', () => {
    void shutdownAllPools().finally(() => process.exit(0));
  });
  
  process.once('SIGINT', () => {
    void shutdownAllPools().finally(() => process.exit(0));
  });
}

/**
 * Gracefully shuts down all active worker pools.
 */
export async function shutdownAllPools(): Promise<void> {
  if (sweepTimer) {
    clearInterval(sweepTimer);
    sweepTimer = undefined;
  }

  const entries = [...sharedWorkerPools.values()];
  sharedWorkerPools.clear();

  await Promise.all(entries.map((entry) => entry.pool.shutdown()));
}

// sourceog-renderer/src/orchestrator/priority-queue.ts
// Alibaba CTO 2027 Standard — Priority-Based Queue

import { LinkedQueue, type QueuedRenderRequest } from './request-queue.js';

/**
 * Priority levels for render requests.
 * Lower numbers = Higher Priority (executed first).
 */
export const enum Priority {
  Critical = 0, // Blocking UI, above the fold, navigation blocking
  High = 1,     // User interaction driven (clicks, form submissions)
  Normal = 2,   // Standard page loads
  Low = 3,      // Background prefetching
  Deferred = 4, // Non-critical, can be delayed indefinitely
}

/**
 * A specialized queue that categorizes requests into priority bands.
 * 
 * Internally uses multiple LinkedQueues (one per priority level) to ensure
 * O(1) operations across the board.
 */
export class PriorityQueue {
  private queues: Map<Priority, LinkedQueue> = new Map();
  private _size = 0;

  constructor() {
    // Initialize queues for all defined priorities
    Object.values(Priority)
      .filter((p): p is Priority => typeof p === 'number')
      .forEach((p) => this.queues.set(p, new LinkedQueue()));
  }

  get size(): number {
    return this._size;
  }

  /**
   * Adds a request to the queue based on its priority.
   */
  enqueue(request: QueuedRenderRequest, priority: Priority = Priority.Normal): void {
    const queue = this.queues.get(priority);
    if (queue) {
      queue.push(request);
      this._size++;
    }
  }

  /**
   * Removes and returns the highest priority request available.
   * Iterates from Priority.Critical (0) upwards.
   */
  dequeue(): QueuedRenderRequest | undefined {
    for (const [_, queue] of this.queues) {
      if (queue.size > 0) {
        const item = queue.shift();
        if (item) {
          this._size--;
          return item;
        }
      }
    }
    return undefined;
  }

  /**
   * Removes a specific request from its priority queue.
   */
  remove(request: QueuedRenderRequest): boolean {
    // We check all queues because we don't store priority on the request object here
    // (though we could optimize by adding priority to the request metadata)
    for (const [_, queue] of this.queues) {
      if (request.node && queue.remove(request.node)) {
        this._size--;
        return true;
      }
    }
    return false;
  }

  /**
   * Drains all queues and returns all requests.
   */
  drain(): QueuedRenderRequest[] {
    const items: QueuedRenderRequest[] = [];
    for (const [_, queue] of this.queues) {
      items.push(...queue.drain());
    }
    this._size = 0;
    return items;
  }
}

// sourceog-renderer/src/orchestrator/request-queue.ts
// Alibaba CTO 2027 Standard — O(1) Doubly Linked Request Queue

import type { WorkerRenderRequest } from '../types/messages.js';

/**
 * Internal payload for a queued render request.
 */
export interface QueuedRenderRequest {
  requestId: string;
  payload: WorkerRenderRequest['payload'];
  resolve: (result: any) => void;
  reject: (error: Error) => void;
  queueTimeout: NodeJS.Timeout;
  renderTimeoutMs: number;
  onChunk?: (chunk: string) => void;
  collectChunks: boolean;
  /** Back-reference to the node for O(1) removal. */
  node?: QueueNode;
}

/**
 * Node for the doubly linked list.
 */
interface QueueNode {
  value: QueuedRenderRequest;
  prev: QueueNode | null;
  next: QueueNode | null;
}

/**
 * High-performance O(1) queue implementation using a doubly linked list.
 * 
 * Why not Array?
 * - Array.shift() is O(n).
 * - Array.splice(index, 1) is O(n).
 * - This implementation ensures constant time for add, remove, and drain operations,
 *   critical for high-throughput worker pool management.
 */
export class LinkedQueue {
  private head: QueueNode | null = null;
  private tail: QueueNode | null = null;
  private _size = 0;

  /**
   * Current number of items in the queue.
   */
  get size(): number {
    return this._size;
  }

  /**
   * Adds a request to the end of the queue.
   * @returns The node reference, needed for O(1) removal later.
   */
  push(value: QueuedRenderRequest): QueueNode {
    const node: QueueNode = { value, prev: this.tail, next: null };
    
    // Store back-reference
    value.node = node;

    if (this.tail) {
      this.tail.next = node;
    } else {
      this.head = node;
    }
    this.tail = node;
    this._size++;
    
    return node;
  }

  /**
   * Removes and returns the request from the front of the queue.
   */
  shift(): QueuedRenderRequest | undefined {
    if (!this.head) return undefined;

    const value = this.head.value;
    this.head = this.head.next;

    if (this.head) {
      this.head.prev = null;
    } else {
      this.tail = null;
    }

    this._size--;
    value.node = undefined; // Clean up reference
    return value;
  }

  /**
   * Removes a specific request from the queue.
   * Used primarily for request cancellation or timeouts.
   * 
   * @param node - The node reference returned by `push`.
   * @returns True if removed, False if not found.
   */
  remove(node: QueueNode): boolean {
    // Validate node is still part of this queue
    if (node.prev === null && node.next === null && this.head !== node) {
      return false; 
    }

    if (node.prev) {
      node.prev.next = node.next;
    } else if (this.head === node) {
      this.head = node.next;
    }

    if (node.next) {
      node.next.prev = node.prev;
    } else if (this.tail === node) {
      this.tail = node.prev;
    }

    node.prev = null;
    node.next = null;
    node.value.node = undefined;
    
    this._size--;
    return true;
  }

  /**
   * Clears the queue and returns all items.
   */
  drain(): QueuedRenderRequest[] {
    const items: QueuedRenderRequest[] = [];
    let current = this.head;
    
    while (current) {
      items.push(current.value);
      current.value.node = undefined; // cleanup
      current = current.next;
    }

    this.head = this.tail = null;
    this._size = 0;
    
    return items;
  }
}

// sourceog-renderer/src/orchestrator/scheduler.ts
// Alibaba CTO 2027 Standard — Worker Scheduling & Dispatch

import type { PriorityQueue, Priority } from './priority-queue.js';
import type { PooledWorker } from './worker-pool.js'; // Forward reference
import type { WorkerRenderRequest, WorkerMessage } from '../types/messages.js';
import type { QueuedRenderRequest } from './request-queue.js';
import { logger } from '../core/logger.js';

/**
 * Interface for the Worker Pool dependency.
 */
export interface WorkerPoolInterface {
  workers: PooledWorker[];
  dispatch(): void;
}

/**
 * The Scheduler is the active agent that moves requests from the Queue to the Workers.
 * 
 * It implements the "Pull" model: whenever a worker becomes idle, the scheduler
 * attempts to pull the highest priority request from the queue and assign it.
 */
export class Scheduler {
  private queue: PriorityQueue;
  private getWorkers: () => PooledWorker[];

  constructor(queue: PriorityQueue, getWorkers: () => PooledWorker[]) {
    this.queue = queue;
    this.getWorkers = getWorkers;
  }

  /**
   * Main dispatch loop.
   * Attempts to assign work to all available idle workers.
   * 
   * @returns The number of workers dispatched.
   */
  public dispatch(): number {
    let dispatchedCount = 0;

    // Continue dispatching as long as we have queue items and idle workers
    while (this.queue.size > 0) {
      const idleWorker = this.findIdleWorker();
      if (!idleWorker) break; // No free workers

      const request = this.queue.dequeue();
      if (!request) break; // Queue empty (race condition safety)

      this.assignWork(idleWorker, request);
      dispatchedCount++;
    }

    return dispatchedCount;
  }

  /**
   * Finds the first available idle worker.
   */
  private findIdleWorker(): PooledWorker | undefined {
    return this.getWorkers().find((w) => !w.busy && !w.stopping);
  }

  /**
   * Assigns a specific render request to a worker.
   */
  private assignWork(worker: PooledWorker, request: QueuedRenderRequest): void {
    // Clear the queue timeout (admission control timeout)
    clearTimeout(request.queueTimeout);

    worker.busy = true;
    worker.requestCount++;
    worker.lastUsed = Date.now();
    worker.current = {
      ...request,
      collectedChunks: [],
      chunkCount: 0,
      renderTimeout: setTimeout(() => {
        // This will trigger the failure handler in the pool
        worker.worker.emit('error', new Error(`Render timeout: ${request.payload.route.id}`));
      }, request.renderTimeoutMs),
    };

    const msg: WorkerRenderRequest = {
      type: 'render',
      requestId: request.requestId,
      payload: request.payload,
    };

    logger.debug(`Dispatching request ${request.requestId} to worker ${worker.worker.threadId}`);
    worker.worker.postMessage(msg);
  }
}

// sourceog-renderer/src/orchestrator/worker-pool.ts
// Alibaba CTO 2027 Standard — Worker Pool Management

import { Worker } from 'node:worker_threads';
import { setTimeout as sleep } from 'node:timers/promises';
import type { WorkerMessage, WorkerRenderRequest } from '../types/messages.js';
import type { WorkerRenderResponse } from '../types/internal.js';
import type { RouteDefinition } from '@sourceog/router';
import type { SourceOGRequestContext } from '@sourceog/runtime';
import type { RenderRouteOptions } from '../types/public.js';
import { PriorityQueue, Priority } from './priority-queue.js';
import { AdmissionController } from './admission-control.js';
import { Scheduler } from './scheduler.js';
import { WorkerSpawnError, RenderError, WorkerQueueTimeoutError } from '@sourceog/genbook/errors';
import { logger } from '../core/logger.js';
import { WORKER_FILE_PATH, MAX_REQUESTS_PER_WORKER, WORKER_TIMEOUT_MS, QUEUE_TIMEOUT_MS, DEFAULT_MAX_QUEUE_DEPTH, DEFAULT_WORKER_COUNT } from '../core/constants.js';
import { toWorkerManifestUrl } from '../core/urls.js';
import { getTranspilerManager } from '../transpiler/transpiler-manager.js';
import { deriveRenderContextKey } from '../core/hashing.js';

/**
 * Represents a worker thread within the pool.
 */
export interface PooledWorker {
  worker: Worker;
  busy: boolean;
  requestCount: number;
  lastUsed: number;
  current?: ActiveRenderRequest;
  stopping: boolean;
}

/**
 * Internal structure for an active render request being processed by a worker.
 */
export interface ActiveRenderRequest {
  requestId: string;
  payload: WorkerRenderRequest['payload'];
  resolve: (result: WorkerRenderResponse) => void;
  reject: (error: Error) => void;
  renderTimeout: NodeJS.Timeout;
  collectedChunks: string[];
  chunkCount: number;
  onChunk?: (chunk: string) => void;
  collectChunks: boolean;
}

/**
 * Configuration options for the WorkerPool.
 */
export interface WorkerPoolOptions {
  workerCount?: number;
  manifestPath?: string;
  maxQueueDepth?: number;
  maxRequestsPerWorker?: number;
}

/**
 * Manages the lifecycle of worker threads, request queuing, and distribution.
 */
export class WorkerPool {
  private readonly workerCount: number;
  private readonly manifestPath?: string;
  private readonly maxRequestsPerWorker: number;
  
  private readonly workers: PooledWorker[] = [];
  private readonly queue: PriorityQueue;
  private readonly admissionController: AdmissionController;
  private readonly scheduler: Scheduler;

  private initialized = false;
  private shuttingDown = false;
  private requestCounter = 0;
  
  // Cached transpiler configuration
  private execArgv: string[] = [];
  private useInlineFallback = false;

  constructor(options: WorkerPoolOptions = {}) {
    this.workerCount = options.workerCount ?? DEFAULT_WORKER_COUNT;
    this.manifestPath = options.manifestPath;
    this.maxRequestsPerWorker = options.maxRequestsPerWorker ?? MAX_REQUESTS_PER_WORKER;
    
    const maxQueueDepth = options.maxQueueDepth ?? DEFAULT_MAX_QUEUE_DEPTH;
    this.queue = new PriorityQueue();
    this.admissionController = new AdmissionController({
      maxQueueDepth,
      maxWorkers: this.workerCount,
    });

    // Scheduler gets access to workers via closure to avoid circular dependency issues
    this.scheduler = new Scheduler(this.queue, () => this.workers);
  }

  /**
   * Initializes the pool by spawning workers.
   */
  public async initialize(): Promise<void> {
    if (this.initialized) return;
    
    const transpiler = getTranspilerManager();
    await transpiler.initialize();
    
    const setup = transpiler.getWorkerExecArgv();
    this.execArgv = setup;
    this.useInlineFallback = transpiler.needsInlineTransform();

    logger.info(`Initializing WorkerPool with ${this.workerCount} workers. Strategy: ${transpiler.getStrategyName()}`);

    await Promise.all(
      Array.from({ length: this.workerCount }, () => this.spawnWorker())
    );

    this.initialized = true;
  }

  /**
   * Renders a route using the worker pool.
   */
  public async render(
    route: RouteDefinition,
    context: SourceOGRequestContext,
    options: RenderRouteOptions = {}
  ): Promise<WorkerRenderResponse> {
    if (this.shuttingDown) {
      throw new RenderError('POOL_SHUTDOWN', 'Pool is shutting down.');
    }

    // 1. Determine Priority (Simplified logic for now)
    const priority = options.onChunk ? Priority.High : Priority.Normal; // Streaming requests are higher priority

    // 2. Admission Control
    const decision = this.admissionController.check(
      this.queue.size,
      this.workers.filter(w => w.busy).length,
      priority
    );

    if (!decision.accepted && decision.error) {
      throw decision.error;
    }

    // 3. Prepare Payload
    const renderContextKey = deriveRenderContextKey(route.pathname, false); // TODO: Handle intercepted
    const payload: WorkerRenderRequest['payload'] = {
      runtimeTarget: 'node', // TODO: Derive from context
      route: {
        id: route.id,
        pathname: route.pathname,
        file: route.file, // Should be URL
        layouts: route.layouts,
      },
      parallelRoutes: {}, // TODO: Handle parallel routes
      context: {
        renderContextKey,
        request: {
          url: context.request.url.toString(),
          method: context.request.method,
          headers: [], // TODO: Serialize headers
          cookies: [], // TODO: Serialize cookies
          requestId: context.request.requestId,
          runtime: context.request.runtime,
        },
        params: context.params,
        query: [...context.query.entries()],
      },
      collectChunks: options.collectChunks ?? false,
      timeoutMs: options.timeoutMs ?? WORKER_TIMEOUT_MS,
    };

    return new Promise((resolve, reject) => {
      const requestId = `rsc-${(++this.requestCounter).toString(36)}`;

      // 4. Queue Request
      const queueTimeout = setTimeout(() => {
        // Note: The queue removal logic is handled inside the queue wrapper or here if we had the node.
        // With Priority Queue, removal on timeout requires scanning or storing reference.
        // For now, we rely on the node reference stored in the request object.
        if (activeReq.node) {
            this.queue.remove(activeReq);
        }
        reject(new WorkerQueueTimeoutError(route.id, QUEUE_TIMEOUT_MS));
      }, QUEUE_TIMEOUT_MS);

      const activeReq: QueuedRenderRequest = {
        requestId,
        payload,
        resolve,
        reject,
        queueTimeout,
        renderTimeoutMs: payload.timeoutMs,
        onChunk: options.onChunk,
        collectChunks: options.collectChunks ?? false,
        // node reference added by queue
      };

      this.queue.enqueue(activeReq, priority);
      
      // 5. Trigger Dispatch
      this.scheduler.dispatch();
    });
  }

  /**
   * Spawns a new worker thread.
   */
  private async spawnWorker(): Promise<PooledWorker> {
    const manifestUrl = toWorkerManifestUrl(this.manifestPath);

    const worker = new Worker(WORKER_FILE_PATH, {
      execArgv: this.execArgv,
      workerData: {
        manifestPath: manifestUrl,
        useInlineTransform: this.useInlineFallback,
      },
      // env: SHARE_ENV is default
    });

    const pooledWorker: PooledWorker = {
      worker,
      busy: false,
      requestCount: 0,
      lastUsed: Date.now(),
      stopping: false,
    };

    worker.on('message', (message: WorkerMessage) => {
      this.handleWorkerMessage(pooledWorker, message);
    });
    
    worker.on('error', (error) => {
      this.handleWorkerFailure(pooledWorker, error);
    });
    
    worker.on('exit', (code) => {
      if (code !== 0 && !pooledWorker.stopping && !this.shuttingDown) {
        this.handleWorkerFailure(pooledWorker, new WorkerSpawnError(`Worker exited with code ${code}`));
      }
    });

    this.workers.push(pooledWorker);
    return pooledWorker;
  }

  /**
   * Handles messages received from a worker.
   */
  private handleWorkerMessage(pw: PooledWorker, message: WorkerMessage): void {
    const current = pw.current;
    if (!current || current.requestId !== message.requestId) return;

    switch (message.type) {
      case 'render_chunk':
        current.chunkCount++;
        if (current.collectChunks) current.collectedChunks.push(message.chunk);
        current.onChunk?.(message.chunk);
        break;

      case 'render_result':
        clearTimeout(current.renderTimeout);
        pw.busy = false;
        pw.current = undefined;
        
        const chunks = message.result.chunks.length > 0 
          ? message.result.chunks 
          : current.collectedChunks;

        current.resolve({
          format: message.result.format,
          chunks,
          streamed: current.chunkCount > 0,
          chunkCount: current.chunkCount,
          usedClientRefs: message.result.usedClientRefs,
        });

        this.checkRecycle(pw);
        this.scheduler.dispatch(); // Trigger next job
        break;

      case 'render_error':
        clearTimeout(current.renderTimeout);
        pw.busy = false;
        pw.current = undefined;
        
        current.reject(new RenderError(message.code, message.error));
        
        this.checkRecycle(pw);
        this.scheduler.dispatch(); // Trigger next job
        break;
    }
  }

  /**
   * Handles unexpected worker failures.
   */
  private async handleWorkerFailure(pw: PooledWorker, error: Error): void {
    logger.error('Worker failure detected', error);
    
    if (pw.current) {
      clearTimeout(pw.current.renderTimeout);
      pw.current.reject(error);
      pw.current = undefined;
      pw.busy = false;
    }

    await this.recycleWorker(pw);
    this.scheduler.dispatch();
  }

  /**
   * Checks if a worker needs to be recycled based on request count.
   */
  private checkRecycle(pw: PooledWorker): void {
    if (pw.requestCount >= this.maxRequestsPerWorker) {
      // Spawn replacement and then stop old one
      void this.recycleWorker(pw);
    }
  }

  /**
   * Replaces a worker with a fresh instance.
   */
  private async recycleWorker(pw: PooledWorker): Promise<void> {
    const idx = this.workers.indexOf(pw);
    if (idx === -1) return;

    // 1. Spawn replacement
    try {
      await this.spawnWorker();
    } catch (err) {
      logger.error('Failed to spawn replacement worker', err);
      return; // Keep old worker if we can't replace?
    }

    // 2. Stop old worker
    this.workers.splice(idx, 1);
    pw.stopping = true;
    try {
      await pw.worker.terminate();
    } catch { /* Ignore */ }
  }

  /**
   * Gracefully shuts down the pool.
   */
  public async shutdown(): Promise<void> {
    this.shuttingDown = true;
    
    // Drain queue
    const pending = this.queue.drain();
    for (const req of pending) {
      clearTimeout(req.queueTimeout);
      req.reject(new RenderError('POOL_SHUTDOWN', 'Pool is shutting down.'));
    }

    // Terminate workers
    await Promise.all(this.workers.map(async (pw) => {
      pw.stopping = true;
      if (pw.current) clearTimeout(pw.current.renderTimeout);
      await pw.worker.terminate();
    }));

    this.workers.length = 0;
    this.initialized = false;
    this.shuttingDown = false;
  }
}

/**
 * Determines caching headers and revalidation strategies for a route.
 * 
 * Supports:
 * - Static caching (Public, immutable)
 * - ISR (Public, revalidate: X)
 * - Dynamic (Private / No-store)
 */
export function resolveCachePolicy(
  route: RouteDefinition,
  context: SourceOGRequestContext
): CachePolicyResult {
  
  // 1. Security Check: Authenticated users should never receive public cached content
  const hasAuth = context.request.headers.has('authorization') || 
                  context.request.cookies.size > 0;

  if (hasAuth) {
    return {
      cache: 'private',
      revalidate: 0, // Do not cache personalized content
      tags: [`user:${context.request.requestId}`], // Scoped tag
    };
  }

  // 2. Route Configuration Check (ISR / Static)
  // Ideally, this reads from `route.config` or similar metadata.
  // Example: if (route.revalidate) return { cache: 'public', revalidate: route.revalidate }

  // 3. Default Policy: Public with short revalidation window (optimistic caching)
  // This allows CDNs to cache while ensuring fresh content eventually.
  const DEFAULT_REVALIDATE = 60; // 1 minute

  return {
    cache: 'public',
    revalidate: DEFAULT_REVALIDATE,
    tags: [route.id, `path:${context.request.url.pathname}`],
  };
}

/**
 * Determines the client-side hydration strategy.
 * 
 * Strategies:
 * - 'eager': Immediately hydrate (default for critical routes).
 * - 'visible': Hydrate when element enters viewport.
 * - 'idle': Hydrate during browser idle time.
 * - 'interaction': Hydrate on user interaction (hover/click).
 */
export function resolveHydrationPolicy(
  route: RouteDefinition,
  context: SourceOGRequestContext
): HydrationPolicyResult {
  
  // 1. Check for explicit route configuration
  // Assume route exports might define a hydration preference
  // e.g., export const config = { hydration: 'lazy' };
  const routeHint = route.config?.hydration;
  if (routeHint) {
    return {
      mode: routeHint as HydrationMode,
    };
  }

  // 2. Heuristics based on context
  const isMobileClient = isMobile(context.request.headers.get('user-agent'));

  // Mobile devices benefit from deferred hydration to preserve resources
  // and improve Time to Interactive (TTI).
  if (isMobileClient) {
    // Defer non-critical hydration to idle time
    return {
      mode: 'idle',
    };
  }

  // 3. Default: Eager Hydration
  // Ensures the app is interactive as soon as possible on desktop/fast connections.
  return {
    mode: 'eager',
  };
}

import type { RouteDefinition } from '@sourceog/router';
import type { SourceOGRequestContext } from '@sourceog/runtime';
import type { ExecutionPlan, ExecutionPlanSegment, RenderStrategy, CachePolicy } from '../types/planning.js';
import { deriveCanonicalRouteId, deriveRenderContextKey } from '../core/hashing.js';
import { resolveRenderPolicy } from './render-policy.js';
import { resolveCachePolicy } from './cache-policy.js';
import { resolveHydrationPolicy } from './hydration-policy.js';

export interface PlannerContext {
  route: RouteDefinition;
  requestContext: SourceOGRequestContext;
  isStaticRoute?: boolean; // Hint from build-time analysis
}

/**
 * The Execution Planner is responsible for determining the optimal rendering
 * strategy for a given route and request context. It bridges the gap between
 * routing and rendering by producing a concrete ExecutionPlan.
 */
export class ExecutionPlanner {
  /**
   * Creates an execution plan for the provided context.
   */
  public plan(ctx: PlannerContext): ExecutionPlan {
    const { route, requestContext, isStaticRoute } = ctx;
    const pathname = requestContext.request.url.pathname;
    
    // 1. Determine the base render policy (SSR, SSG, ISR)
    const renderPolicy = resolveRenderPolicy(route, requestContext, isStaticRoute);
    
    // 2. Determine caching rules
    const cachePolicy = resolveCachePolicy(route, requestContext);
    
    // 3. Determine client-side hydration strategy
    const hydrationPolicy = resolveHydrationPolicy(route, requestContext);

    // 4. Build the segment graph
    const segments = this.buildSegments(route, renderPolicy.strategy, cachePolicy);

    // 5. Construct the final plan
    return {
      routeId: route.id,
      pathname,
      shell: {
        mode: renderPolicy.shellMode,
        cache: cachePolicy.cache,
        revalidate: cachePolicy.revalidate,
      },
      segments,
      hydration: {
        mode: hydrationPolicy.mode,
      },
    };
  }

  /**
   * Constructs the list of render segments for the execution plan.
   * In a full implementation, this would analyze the component tree.
   */
  private buildSegments(
    route: RouteDefinition,
    strategy: RenderStrategy,
    cache: CachePolicy
  ): ExecutionPlanSegment[] {
    const segments: ExecutionPlanSegment[] = [];
    let priority = 10;

    // Layouts (Higher priority)
    for (const layoutFile of route.layouts) {
      segments.push({
        id: `layout:${layoutFile}`,
        kind: 'layout',
        strategy: 'stream', // Layouts usually stream
        priority: priority++,
        cache,
      });
    }

    // Template (if present)
    if (route.templateFile) {
      segments.push({
        id: `template:${route.templateFile}`,
        kind: 'layout',
        strategy: 'stream',
        priority: priority++,
        cache,
      });
    }

    // Page (Core content)
    segments.push({
      id: `page:${route.file}`,
      kind: 'page',
      strategy,
      priority: priority,
      cache,
      timeoutMs: 5000, // Default page timeout
    });

    return segments;
  }
}

// sourceog-renderer/src/planning/prefetch-policy.ts
// Alibaba CTO 2027 Standard — Prefetch Policy Resolution

import type { RouteDefinition } from '@sourceog/router';
import type { SourceOGRequestContext } from '@sourceog/runtime';

/**
 * The strategy to use for prefetching linked resources.
 * - 'intent': Prefetch on hover/focus (optimal balance).
 * - 'viewport': Prefetch when links enter viewport (aggressive).
 * - 'render': Prefetch immediately after page load (very aggressive).
 * - 'none': Disable prefetching.
 */
export type PrefetchStrategy = 'intent' | 'viewport' | 'render' | 'none';

export interface PrefetchPolicyResult {
  strategy: PrefetchStrategy;
  /** 
   * Specific relative paths to prefetch. 
   * If empty, the client runtime will heuristically scan the DOM for <a> tags.
   */
  targets?: string[];
}

/**
 * Determines the prefetching strategy based on network conditions and route priority.
 * 
 * Optimizations:
 * - Respects the `Save-Data` Client Hint header.
 * - Reduces aggression on mobile networks.
 * - Disables for high-latency contexts.
 */
export function resolvePrefetchPolicy(
  route: RouteDefinition,
  context: SourceOGRequestContext
): PrefetchPolicyResult {
  
  // 1. Check for Data Saver Mode ( browsers send 'Save-Data: on')
  const saveData = context.request.headers.get('save-data');
  if (saveData === 'on') {
    return { strategy: 'none' };
  }

  // 2. Effective Connection Type (ECT) Heuristic
  // Note: Headers like 'ect' are part of Client Hints (Network Information API)
  const ect = context.request.headers.get('ect'); // '4g', '3g', '2g', 'slow-2g'
  if (ect === '2g' || ect === 'slow-2g') {
    return { strategy: 'none' };
  }

  // 3. Default Strategy: Intent-based Prefetching
  // This provides the best balance between perceived performance and bandwidth usage.
  // It only loads resources the user is actively interested in.
  return {
    strategy: 'intent',
    // targets could be populated here if we had a static graph of likely next routes
  };

// sourceog-renderer/src/planning/render-policy.ts
// Alibaba CTO 2027 Standard — Render Policy Resolution

import type { RouteDefinition } from '@sourceog/router';
import type { SourceOGRequestContext } from '@sourceog/runtime';
import type { RenderPolicyResult } from './planner-types.js';
import type { RenderStrategy } from '../types/planning.js';

/**
 * Determines the optimal rendering strategy (SSG, ISR, SSR, Stream) 
 * for a given route and request context.
 * 
 * Priority:
 * 1. Force-dynamic headers.
 * 2. Static generation hints (build-time).
 * 3. Revalidation configuration (ISR).
 * 4. Default to Streaming (RSC).
 */
export function resolveRenderPolicy(
  route: RouteDefinition,
  context: SourceOGRequestContext,
  isStaticHint?: boolean
): RenderPolicyResult {
  
  // 1. Check for Force-Dynamic (Skip caching/static gen)
  if (isForceDynamic(context)) {
    return {
      strategy: 'stream', // Always stream dynamic content
      shellMode: 'dynamic',
    };
  }

  // 2. Static Generation (SSG)
  if (isStaticHint) {
    // Check for ISR configuration
    const revalidate = route.revalidate; // Assuming route definition has this
    
    if (typeof revalidate === 'number' && revalidate > 0) {
      return {
        strategy: 'isr',
        shellMode: 'static',
      };
    }
    
    return {
      strategy: 'ssg',
      shellMode: 'static',
    };
  }

  // 3. Default for RSC: Streaming SSR
  // This allows progressive delivery of the shell and chunks.
  return {
    strategy: 'stream',
    shellMode: 'dynamic',
  };
}

/**
 * Checks if the request forces a dynamic render.
 */
function isForceDynamic(context: SourceOGRequestContext): boolean {
  const headers = context.request.headers;
  
  // Check for standard cache control headers
  const cacheControl = headers.get('cache-control');
  if (cacheControl?.includes('no-store') || cacheControl?.includes('no-cache')) {
    return true;
  }
  
  // Check for preview/preview mode headers (common in CMS integration)
  if (headers.get('x-preview-mode') === 'true') {
    return true;
  }

  return false;
}

// sourceog-renderer/src/planning/segment-graph.ts
// Alibaba CTO 2027 Standard — Segment Graph Construction

import type { RouteDefinition } from '@sourceog/router';
import type { ExecutionPlanSegment, RenderStrategy, CachePolicy } from '../types/planning.js';

/**
 * Represents a node in the render segment tree.
 */
export interface SegmentNode {
  id: string;
  type: 'layout' | 'template' | 'page' | 'parallel';
  filePath: string;
  children: SegmentNode[];
}

/**
 * Constructs the segment graph for a given route.
 * This graph dictates the order and nesting of rendering operations.
 * 
 * Root -> Layout -> Template -> Page
 *                     |
 *                   Parallel Slots
 */
export function buildSegmentGraph(route: RouteDefinition): SegmentNode {
  const rootNode: SegmentNode = {
    id: 'root',
    type: 'layout', // Virtual root
    filePath: '',
    children: [],
  };

  let cursor = rootNode;

  // 1. Add Layout Segments (Top-down order)
  for (const layoutFile of route.layouts) {
    const layoutNode: SegmentNode = {
      id: `layout:${layoutFile}`,
      type: 'layout',
      filePath: layoutFile,
      children: [],
    };
    cursor.children.push(layoutNode);
    cursor = layoutNode;
  }

  // 2. Add Template Segment (if exists)
  if (route.templateFile) {
    const templateNode: SegmentNode = {
      id: `template:${route.templateFile}`,
      type: 'template',
      filePath: route.templateFile,
      children: [],
    };
    cursor.children.push(templateNode);
    cursor = templateNode;
  }

  // 3. Add Page Segment
  const pageNode: SegmentNode = {
    id: `page:${route.file}`,
    type: 'page',
    filePath: route.file,
    children: [],
  };
  cursor.children.push(pageNode);

  // 4. Add Parallel Routes (Slots) - attached to the same cursor as Page
  // Note: In RSC, parallel routes are rendered alongside the page.
  if (route.parallelRoutes) {
    for (const [slot, slotRoute] of Object.entries(route.parallelRoutes)) {
      const parallelNode: SegmentNode = {
        id: `parallel:${slot}:${slotRoute.file}`,
        type: 'parallel',
        filePath: slotRoute.file,
        children: [],
      };
      // Parallel routes are siblings to the page, attached to the parent (template/layout)
      cursor.children.push(parallelNode);
    }
  }

  return rootNode;
}

/**
 * Flattens the segment graph into a linear execution list.
 * Order is determined by render priority (Layouts first, then Page).
 */
export function flattenSegmentGraph(
  root: SegmentNode,
  defaultStrategy: RenderStrategy,
  defaultCache: CachePolicy
): ExecutionPlanSegment[] {
  const segments: ExecutionPlanSegment[] = [];
  
  // DFS traversal
  function visit(node: SegmentNode, depth: number) {
    if (node.type !== 'root') {
      segments.push({
        id: node.id,
        kind: node.type === 'parallel' ? 'deferred' : node.type,
        strategy: defaultStrategy,
        priority: 100 - depth, // Higher depth = lower priority (layouts have high priority)
        cache: defaultCache,
      });
    }
    
    for (const child of node.children) {
      visit(child, depth + 1);
    }
  }

  visit(root, 0);
  return segments;
}

// sourceog-renderer/src/planning/static-analyzer.ts
// Alibaba CTO 2027 Standard — Route Static Analysis

import type { StaticAnalysisResult } from './planner-types.js';
import { logger } from '../core/logger.js';
import { readFileSync, existsSync } from 'node:fs';
import path from 'node:path';

/**
 * Analyzes route modules to determine static eligibility and dynamic behaviors.
 * 
 * This implementation uses lightweight heuristics (export inspection) and 
 * simple source scanning. For production build-time analysis, this should be
 * replaced with a full AST-based analyzer.
 */
export class StaticAnalyzer {
  /**
   * Analyzes a route module to determine its rendering characteristics.
   * 
   * @param moduleExports - The imported module object to inspect.
   * @param filePath - The file path (used for source scanning if needed).
   */
  public analyze(
    moduleExports: Record<string, unknown>,
    filePath?: string
  ): StaticAnalysisResult {
    const result: StaticAnalysisResult = {
      isPure: true, // Assume pure until proven dynamic
      hasActions: false,
      clientBoundaries: [],
    };

    // 1. Check for dynamic function exports (config hints)
    // If a route exports specific configs, it might opt-in to dynamic.
    
    // 2. Inspect exported functions
    if (typeof moduleExports.generateStaticParams === 'function') {
      // Presence of generateStaticParams implies the route *can* be static,
      // but we might still want to render it dynamically on fallback.
      // It does not immediately flag isPure = false.
    }

    // 3. Inspect for Server Actions
    // Actions usually force dynamic rendering contexts but not necessarily the whole route.
    // We scan for common action naming patterns or metadata.
    for (const key of Object.keys(moduleExports)) {
      const exportValue = moduleExports[key];
      
      // Heuristic: functions starting with uppercase are usually components,
      // lowercase are usually utility or actions.
      // A more robust check involves checking if the function is bound to a specific action context.
      if (typeof exportValue === 'function' && key.startsWith('action')) {
        result.hasActions = true;
      }
    }

    // 4. Source Code Scanning (Heuristic)
    // If source is available, scan for dynamic runtime hooks.
    if (filePath && existsSync(filePath)) {
      try {
        const source = readFileSync(filePath, 'utf-8');
        
        // Detection patterns for dynamic rendering triggers
        const dynamicPatterns = [
          /cookies\s*\(\s*\)/,          // cookies()
          /headers\s*\(\s*\)/,          // headers()
          /searchParams/,               // usage of searchParams prop
          /Math\.random\s*\(\s*\)/,     // random numbers
          /Date\.now\s*\(\s*\)/,        // current time (side effects)
        ];

        for (const pattern of dynamicPatterns) {
          if (pattern.test(source)) {
            result.isPure = false;
            logger.debug(`Static analysis detected dynamic pattern in ${filePath}: ${pattern}`);
            break;
          }
        }
      } catch (err) {
        // Ignore read errors; rely on exports analysis
      }
    }

    return result;
  }
}

// Singleton instance
export const staticAnalyzer = new StaticAnalyzer();

import type { ServerResponse } from 'node:http';
import { PassThrough } from 'node:stream';

/**
 * The official MIME type for React Server Components payload.
 */
export const RSC_CONTENT_TYPE = 'text/x-component';

/**
 * Headers required for a valid RSC response.
 */
export const RSC_HEADERS = {
  'Content-Type': RSC_CONTENT_TYPE,
  'X-SourceOG-RSC': '1',
  // Connection: keep-alive is usually handled by the server, but good to note
};

/**
 * Checks if a request accepts RSC content.
 * @param acceptHeader - The value of the 'Accept' header.
 */
export function acceptsFlight(acceptHeader: string | null | undefined): boolean {
  if (!acceptHeader) return false;
  return acceptHeader.includes(RSC_CONTENT_TYPE) || acceptHeader.includes('*/*');
}

/**
 * Pipes a flight stream to a ServerResponse with appropriate headers.
 * Handles backpressure and error propagation.
 * 
 * @param stream - The PassThrough stream containing the RSC payload.
 * @param res - The Node.js ServerResponse object.
 */
export function pipeFlightToResponse(stream: PassThrough, res: ServerResponse): void {
  // Set Headers
  Object.entries(RSC_HEADERS).forEach(([key, value]) => {
    res.setHeader(key, value);
  });

  // Handle Errors
  stream.on('error', (err) => {
    console.error('[RSC Flight] Stream error:', err);
    if (!res.headersSent) {
      res.statusCode = 500;
      res.end('RSC Stream Error');
    } else {
      res.destroy();
    }
  });

  // Pipe
  stream.pipe(res);
}

/**
 * Decodes a simple text chunk into a string.
 * Used for collecting chunks for SSR rendering of the HTML shell.
 */
export function decodeChunk(chunk: Buffer | string): string {
  return typeof chunk === 'string' ? chunk : chunk.toString('utf-8');
}

// Configuration passed from the worker entry point
let USE_INLINE_TRANSFORM = false;

/**
 * Configures the module loader for the current worker context.
 * Must be called once during worker initialization.
 */
export function configureModuleLoader(options: { useInlineTransform: boolean }): void {
  USE_INLINE_TRANSFORM = options.useInlineTransform;
}

/**
 * Loads a route module (Page, Layout, Template) dynamically.
 * Handles both direct JS imports and TS/TSX transpilation via the UTAL layer.
 * 
 * @param specifier - The file path or URL to the route module.
 * @returns The module exports.
 */
export async function loadRouteModule(
  specifier: string
): Promise<{ default?: unknown; [key: string]: unknown }> {
  let moduleUrl: string;
  
  try {
    // Normalize specifier to URL
    moduleUrl = specifier.startsWith('file://') 
      ? specifier 
      : pathToFileURL(path.resolve(specifier)).href;

    // Delegate to the UTAL loader
    const mod = await loadModule(moduleUrl, USE_INLINE_TRANSFORM);
    
    return mod as { default?: unknown; [key: string]: unknown };
  } catch (error) {
    const fsPath = specifier.startsWith('file://') ? fileURLToPath(specifier) : specifier;
    
    logger.error(`Failed to load route module: ${fsPath}`, error);
    
    if (error instanceof TranspileError || error instanceof ModuleLoadError) {
      throw error;
    }
    
    throw new ModuleLoadError(specifier, fsPath, error);
  }
}

export function getComponentFromModule(
  mod: { default?: unknown; [key: string]: unknown },
  exportName: string = 'default'
): unknown {
  if (exportName === 'default') {
    return mod.default;
  }
  return mod[exportName];

// sourceog-renderer/src/rsc/compat-renderer.ts
// Alibaba CTO 2027 Standard — RSC Compatibility Layer (Renderer)

import React from 'react';
import { PassThrough } from 'node:stream';
import { RenderError, RenderTimeoutError } from '@sourceog/genbook/errors';
import { logger } from '../core/logger.js';

// ---------------------------------------------------------------------------
// Dynamic Imports for React Server Dom
// ---------------------------------------------------------------------------

// We lazy load the renderer to avoid loading it in environments that don't use RSC.
type RenderToPipeableStreamFn = (
  model: unknown,
  webpackMap: unknown,
  options?: { onError?: (error: unknown) => void }
) => {
  pipe: (destination: NodeJS.WritableStream) => NodeJS.WritableStream;
  abort?: (reason?: unknown) => void;
};

let _renderToPipeableStream: RenderToPipeableStreamFn | null = null;

async function getRenderer(): Promise<RenderToPipeableStreamFn> {
  if (_renderToPipeableStream) return _renderToPipeableStream;

  try {
    // @ts-ignore - React Server DOM Webpack is not typed in standard React types
    const mod = await import('react-server-dom-webpack/server.node');
    _renderToPipeableStream = mod.renderToPipeableStream;
    return _renderToPipeableStream!;
  } catch (error) {
    logger.error('Failed to load react-server-dom-webpack. Ensure "react-server-dom-webpack" is installed.', error);
    throw new RenderError(
      'RENDERER_UNAVAILABLE', 
      'RSC Renderer (react-server-dom-webpack) is not available.'
    );
  }
}

// ---------------------------------------------------------------------------
// Public API: renderFlight
// ---------------------------------------------------------------------------

export interface RenderFlightOptions {
  timeoutMs: number;
  routeId: string;
}

/**
 * Renders a React element tree to a React Flight (RSC) stream.
 * Handles timeout, error boundaries, and stream collection.
 * 
 * @param element - The root React element.
 * @param manifest - The normalized client reference manifest.
 * @param options - Rendering options (timeout, context).
 * @returns A promise resolving to the collected chunks.
 */
export async function renderFlight(
  element: React.ReactElement,
  manifest: Record<string, unknown>,
  options: RenderFlightOptions
): Promise<string[]> {
  const renderToPipeableStream = await getRenderer();
  
  return new Promise((resolve, reject) => {
    const chunks: string[] = [];
    const stream = new PassThrough();
    
    // 1. Setup Timeout
    const timeout = setTimeout(() => {
      stream.destroy();
      reject(new RenderTimeoutError(options.routeId, options.timeoutMs));
    }, options.timeoutMs);

    // 2. Handle Stream Events
    stream.on('data', (chunk: Buffer | string) => {
      chunks.push(chunk.toString('utf8'));
    });

    stream.on('error', (err) => {
      clearTimeout(timeout);
      reject(new RenderError('STREAM_ERROR', err.message, options.routeId));
    });

    stream.on('end', () => {
      clearTimeout(timeout);
      resolve(chunks);
    });

    // 3. Execute Render
    try {
      const flightStream = renderToPipeableStream(element, manifest, {
        onError(error) {
          clearTimeout(timeout);
          const msg = error instanceof Error ? error.message : String(error);
          // Reject promise if the render fails before streaming starts
          reject(new RenderError('RENDER_ERROR', msg, options.routeId));
        },
      });

      flightStream.pipe(stream);
    } catch (err) {
      clearTimeout(timeout);
      reject(new RenderError('SETUP_ERROR', (err as Error).message, options.routeId));
    }
  });
}

// sourceog-renderer/src/rsc/compat-route-loader.ts

import type { WorkerRouteDefinition } from '../types/internal.js';
import { loadRouteModule } from './compat-module-loader.js';
import { RenderError } from '@sourceog/genbook/errors';

/**
 * Represents a loaded route module with resolved component references.
 */
export interface LoadedRouteComponent {
  id: string;
  pathname: string;
  /** The main component to render (default export or 'Page' export). */
  component: unknown;
  /** Loaded layout components (nested order). */
  layouts: unknown[];
  /** Loaded template component (if present). */
  template: unknown | null;
}

/**
 * Loads the route definition and resolves the React components necessary for rendering.
 * Handles the "Alibaba CTO 2027" convention:
 * - `default` export is the Page.
 * - `Page` export is also checked.
 * - Layouts are `default` exports from layout files.
 */
export async function resolveRouteComponents(
  route: WorkerRouteDefinition
): Promise<LoadedRouteComponent> {
  // 1. Load Page Component
  const pageModule = await loadRouteModule(route.file);
  
  // Convention: Prefer `default`, fallback to `Page` named export
  const pageComponent = pageModule.default ?? (pageModule as any).Page;
  
  if (typeof pageComponent !== 'function') {
    throw new RenderError(
      'INVALID_ROUTE_MODULE',
      `Route "${route.id}" does not export a valid React component.`,
      route.id
    );
  }

  // 2. Load Layout Components
  const layouts: unknown[] = [];
  for (const layoutFile of route.layouts) {
    const layoutModule = await loadRouteModule(layoutFile);
    const layoutComponent = layoutModule.default;
    
    if (typeof layoutComponent !== 'function') {
      console.warn(`[SOURCEOG] Layout "${layoutFile}" does not have a valid default export.`);
      continue;
    }
    layouts.push(layoutComponent);
  }

  // 3. Load Template Component (if present)
  let template: unknown | null = null;
  if (route.templateFile) {
    const templateModule = await loadRouteModule(route.templateFile);
    template = templateModule.default;
    if (typeof template !== 'function') {
      console.warn(`[SOURCEOG] Template "${route.templateFile}" is not a valid component.`);
      template = null;
    }
  }

  return {
    id: route.id,
    pathname: route.pathname,
    component: pageComponent,
    layouts,
    template,
  };
}

// sourceog-renderer/src/transpiler/transpiler-core.ts
// Alibaba CTO 2027 Standard — Unified Transpiler Abstraction Layer (UTAL)

import { createRequire } from 'node:module';
import path from 'node:path';
import { pathToFileURL, fileURLToPath } from 'node:url';
import { existsSync, readFileSync } from 'node:fs';
import { 
  TranspileError, 
  NoTranspilerError, 
  TransformStrategyError 
} from '@sourceog/genbook/errors';
import { 
  isBun, 
  isDeno, 
  isNode, 
  RUNTIME_MAJOR, 
  isProduction 
} from '@sourceog/genbook/utils';
import { logger } from '../core/logger.js';

// ---------------------------------------------------------------------------
// Types
// ---------------------------------------------------------------------------

export type TranspilerStrategy = 
  | 'bun-native'
  | 'deno-native'
  | 'node-strip-types'
  | 'node-transform-types'
  | 'tsx'
  | 'esbuild'
  | 'sucrase'
  | 'none';

export interface TranspilerDetectionResult {
  strategy: TranspilerStrategy;
  execArgv: string[];
  useInlineFallback: boolean;
}

// ---------------------------------------------------------------------------
// Transpiler Manager
// ---------------------------------------------------------------------------

class TranspilerManager {
  private detectionResult: TranspilerDetectionResult | null = null;
  private initPromise: Promise<void> | null = null;

  // Inline transform caches
  private esbuildTransform: ((code: string, filename: string) => Promise<string>) | null = null;
  private sucraseTransform: ((code: string, filename: string) => Promise<string>) | null = null;

  /**
   * Detects the best available transpiler strategy for the current environment.
   * Must be called once at startup.
   */
  public async initialize(): Promise<void> {
    if (this.detectionResult) return;
    if (this.initPromise) return this.initPromise;

    this.initPromise = this.detect().finally(() => {
      this.initPromise = null;
    });

    return this.initPromise;
  }

  private async detect(): Promise<void> {
    let strategy: TranspilerStrategy = 'none';
    let execArgv: string[] = [];
    let useInlineFallback = false;

    // 1. Bun / Deno Native
    if (isBun) {
      strategy = 'bun-native';
      this.detectionResult = { strategy, execArgv: [], useInlineFallback: false };
      return;
    }
    if (isDeno) {
      strategy = 'deno-native';
      this.detectionResult = { strategy, execArgv: [], useInlineFallback: false };
      return;
    }

    // 2. Node.js Native Transform Types (v22.6+)
    if (isNode && RUNTIME_MAJOR >= 22) {
      strategy = 'node-transform-types';
      execArgv = ['--experimental-transform-types', '--conditions=react-server'];
      this.detectionResult = { strategy, execArgv, useInlineFallback: false };
      return;
    }

    // 3. TSX Loader (Preferred External Loader)
    const tsxLoader = await this.findTsxLoader();
    if (tsxLoader) {
      strategy = 'tsx';
      execArgv = ['--import', tsxLoader, '--conditions=react-server'];
      this.detectionResult = { strategy, execArgv, useInlineFallback: false };
      return;
    }

    // 4. Fallback to Inline Transform (Esbuild/Sucrase)
    logger.warn(
      '[SOURCEOG] No native transpiler or tsx found. Falling back to inline transformation. ' +
      'Install "tsx" or "esbuild" for better performance.'
    );

    await this.setupInlineTransformers();

    if (this.esbuildTransform) {
      strategy = 'esbuild';
      useInlineFallback = true;
    } else if (this.sucraseTransform) {
      strategy = 'sucrase';
      useInlineFallback = true;
    } else {
      // 5. No Transpiler Available
      throw new NoTranspilerError('.tsx', ['native', 'tsx', 'esbuild', 'sucrase']);
    }

    this.detectionResult = { 
      strategy, 
      execArgv: ['--conditions=react-server'], 
      useInlineFallback 
    };
  }

  /**
   * Returns the execArgv needed for worker threads.
   */
  public getWorkerExecArgv(): string[] {
    return this.detectionResult?.execArgv ?? [];
  }

  /**
   * Returns true if the worker needs to perform its own transpilation.
   */
  public useInlineTransform(): boolean {
    return this.detectionResult?.useInlineFallback ?? false;
  }

  /**
   * Transforms source code inline (Main Thread Fallback).
   */
  public async transform(code: string, filename: string): Promise<string> {
    if (this.esbuildTransform) {
      try {
        return await this.esbuildTransform(code, filename);
      } catch (err) {
        throw new TransformStrategyError('esbuild', filename, err);
      }
    }

    if (this.sucraseTransform) {
      try {
        return await this.sucraseTransform(code, filename);
      } catch (err) {
        throw new TransformStrategyError('sucrase', filename, err);
      }
    }

    throw new NoTranspilerError(path.extname(filename), ['esbuild', 'sucrase']);
  }

  // ---------------------------------------------------------------------------
// Private Helpers
// ---------------------------------------------------------------------------

  private async setupInlineTransformers(): Promise<void> {
    // Try Esbuild
    try {
      const esbuild = await import('esbuild');
      this.esbuildTransform = async (code, filename) => {
        const result = await esbuild.transform(code, {
          loader: filename.endsWith('.tsx') ? 'tsx' : filename.endsWith('.jsx') ? 'jsx' : 'ts',
          jsx: 'automatic',
          jsxImportSource: 'react',
          target: 'es2022',
          format: 'esm',
        });
        return result.code;
      };
      return;
    } catch { /* Fallthrough */ }

    // Try Sucrase
    try {
      const sucrase = await import('sucrase');
      this.sucraseTransform = async (code, filename) => {
        return sucrase.transform(code, {
          transforms: ['typescript', 'jsx'],
          jsxRuntime: 'automatic',
          filePath: filename,
        }).code;
      };
    } catch { /* Fallthrough */ }
  }

  private async findTsxLoader(): Promise<string | null> {
    try {
      // Strategy 1: import.meta.resolve
      // @ts-ignore
      if (typeof import.meta.resolve === 'function') {
        // @ts-ignore
        const url = import.meta.resolve('tsx/esm');
        if (url) return url;
      }
    } catch {}

    // Strategy 2: Package.json exports lookup from CWD
    const req = createRequire(path.join(process.cwd(), 'dummy.js'));
    try {
      const pkgPath = req.resolve('tsx/package.json');
      const pkg = JSON.parse(readFileSync(pkgPath, 'utf8'));
      const esmExport = pkg.exports?.['./esm'];
      let rel: string | undefined;
      
      if (typeof esmExport === 'string') rel = esmExport;
      else if (typeof esmExport === 'object') rel = esmExport?.import ?? esmExport?.default;

      if (rel) {
        const resolved = path.join(path.dirname(pkgPath), rel);
        if (existsSync(resolved)) return pathToFileURL(resolved).href;
      }
    } catch {}

    return null;
  }
}

/**
 * Loads a module inside a worker, handling inline transpilation if necessary.
 * 
 * @param specifier - The file path or URL to import.
 * @param useInlineTransform - Whether to attempt transpilation for TS/TSX files.
 * @returns The module exports.
 */
export async function loadModule(
  specifier: string,
  useInlineTransform: boolean = false
): Promise {
  const fsPath = specifier.startsWith('file://') ? fileURLToPath(specifier) : specifier;
  const ext = path.extname(fsPath);

  // 1. No transform needed for standard JS
  if (ext === '.js' || ext === '.mjs' || ext === '.cjs') {
    return import(specifier.startsWith('file://') ? specifier : pathToFileURL(fsPath).href);
  }

  // 2. Check if transform is needed/supported
  const needsTransform = ext === '.tsx' || ext === '.ts' || ext === '.jsx';
  if (!needsTransform) {
    // Try native import for unknown extensions
    return import(specifier.startsWith('file://') ? specifier : pathToFileURL(fsPath).href);
  }

  if (!useInlineTransform) {
    // If not using inline transform, rely on native Node flags (already set in execArgv)
    // But if we reach here and it's TSX, something is wrong or we should try native.
    return import(specifier.startsWith('file://') ? specifier : pathToFileURL(fsPath).href);
  }

  // 3. Perform Inline Transform
  return loadWithInlineTransform(fsPath);
}

/**
 * Transforms source and loads from a temporary file to support relative imports.
 */
async function loadWithInlineTransform(fsPath: string): Promise {
  const source = readFileSync(fsPath, 'utf8');
  const sourceHash = createHash('sha256').update(source).digest('hex').slice(0, 8);
  
  // Create a temp file path preserving the filename structure for debug
  const tmpFile = path.join(
    TRANSFORM_TMP_DIR, 
    `${path.basename(fsPath).replace(/[^a-zA-Z0-9.]/g, '_')}-${sourceHash}.mjs`
  );

  // Check cache (file existence)
  if (existsSync(tmpFile)) {
    return import(pathToFileURL(tmpFile).href);
  }

  // Determine transformer
  let transformFn = await getEsbuildTransform();
  if (!transformFn) {
    transformFn = await getSucraseTransform();
  }

  if (!transformFn) {
    throw new TranspileError(
      'NO_INLINE_TRANSFORMER',
      'No inline transformer (esbuild/sucrase) found for worker bootstrap.',
      fsPath
    );
  }

  try {
    const transformedCode = await transformFn(source, fsPath);
    writeFileSync(tmpFile, transformedCode, 'utf8');
    return import(pathToFileURL(tmpFile).href);
  } catch (err) {
    throw new TransformStrategyError('inline-transform', fsPath, err);
  }
}

// ---------------------------------------------------------------------------
// Utility: resolveImportSpecifier
// ---------------------------------------------------------------------------

/**
 * Resolves an import specifier relative to a parent path.
 */
export function resolveImportSpecifier(specifier: string, parentPath: string): string {
  if (!specifier || typeof specifier !== 'string') {
    throw new TypeError(`Invalid import specifier: ${specifier}`);
  }
  
  if (specifier.startsWith('node:') || specifier.startsWith('data:') || specifier.startsWith('file://')) {
    return specifier;
  }
  
  if (path.isAbsolute(specifier)) {
    return pathToFileURL(specifier).href;
  }

  // Relative or bare specifier
  if (specifier.startsWith('.')) {
    const parentDir = path.dirname(parentPath);
    const resolved = path.resolve(parentDir, specifier);
    return pathToFileURL(resolved).href;
  }

  // Bare specifier - try to resolve via require
  try {
    const req = createRequire(parentPath);
    const resolved = req.resolve(specifier);
    return pathToFileURL(resolved).href;
  } catch {
    // Return as-is if cannot resolve (might be handled by loader)
    return specifier;
  }
}

import type { Worker } from 'node:worker_threads';
import { logger } from '../core/logger.js';

export interface HealthCheckResult {
  threadId: number;
  isAlive: boolean;
  lastPong: number | null;
  memoryUsage?: NodeJS.MemoryUsage;
}

/**
 * Monitors the health of worker threads.
 * Implements heartbeat (ping/pong) and memory watchdogs.
 */
export class WorkerHealthMonitor {
  private lastPongTimestamps = new Map<number, number>();
  private pingIntervalId: NodeJS.Timeout | null = null;

  /**
   * Starts the periodic health check loop.
   */
  public start(workers: Worker[]): void {
    if (this.pingIntervalId) return;

    this.pingIntervalId = setInterval(() => {
      this.runCheck(workers);
    }, 30_000); // Check every 30 seconds

    if (this.pingIntervalId.unref) {
      this.pingIntervalId.unref();
    }
  }

  /**
   * Stops the health check loop.
   */
  public stop(): void {
    if (this.pingIntervalId) {
      clearInterval(this.pingIntervalId);
      this.pingIntervalId = null;
    }
    this.lastPongTimestamps.clear();
  }

  /**
   * Records a pong response from a worker.
   */
  public recordPong(threadId: number): void {
    this.lastPongTimestamps.set(threadId, Date.now());
  }

  /**
   * Runs a single health check pass.
   */
  private runCheck(workers: Worker[]): void {
    const now = Date.now();

    for (const worker of workers) {
      // 1. Send Ping
      worker.postMessage({ type: 'health_ping' });

      // 2. Check Memory (Watchdog)
      // Note: worker.getHeapStatistics() is asynchronous and specific to Node versions.
      // In production, we often rely on external monitoring or process-level checks.
      
      // 3. Check Responsiveness (based on last pong)
      const lastPong = this.lastPongTimestamps.get(worker.threadId);
      
      // If no pong received within 60 seconds, worker might be stuck
      if (lastPong && (now - lastPong > 60_000)) {
        logger.warn(`Worker ${worker.threadId} has not responded to health check in 60s.`);
        // Trigger recycling logic (handled by pool)
      }
    }
  }

  /**
   * Instantiates a health check result for a specific worker.
   */
  public checkHealth(worker: Worker): HealthCheckResult {
    const lastPong = this.lastPongTimestamps.get(worker.threadId) ?? null;
    
    return {
      threadId: worker.threadId,
      // Basic check: if we have a recent pong or the worker is new
      isAlive: lastPong ? (Date.now() - lastPong < 60_000) : true, 
      lastPong,
    };
  }
}

type LifecycleState = 'initializing' | 'ready' | 'busy' | 'shutting_down' | 'dead';

/**
 * Manages the lifecycle state of the worker thread.
 * Used to gate operations during startup and shutdown.
 */
export class WorkerLifecycle {
  private state: LifecycleState = 'initializing';
  private port: MessagePort | null = null;

  constructor() {}

  /**
   * Marks the worker as ready to accept render requests.
   */
  public ready(port: MessagePort): void {
    this.port = port;
    this.state = 'ready';
    logger.debug('Worker lifecycle: READY');
  }

  /**
   * Marks the worker as busy rendering.
   */
  public beginRender(): boolean {
    if (this.state !== 'ready') {
      return false; // Cannot accept render
    }
    this.state = 'busy';
    return true;
  }

  /**
   * Marks the worker as available again.
   */
  public endRender(): void {
    if (this.state === 'shutting_down') return;
    this.state = 'ready';
  }

  /**
   * Initiates graceful shutdown.
   */
  public shutdown(): void {
    if (this.state === 'dead') return;
    
    this.state = 'shutting_down';
    logger.debug('Worker lifecycle: SHUTTING_DOWN');
    
    // Flush any pending metrics or logs if needed
    
    this.state = 'dead';
    logger.debug('Worker lifecycle: DEAD');
  }

  /**
   * Checks if the worker can accept work.
   */
  public isAcceptingWork(): boolean {
    return this.state === 'ready';
  }
}

// Alibaba CTO 2027 Standard — Worker Request Router

import type { MessagePort } from 'node:worker_threads';
import type { WorkerRenderRequest, WorkerMessage } from '../types/messages.js';
import type { NormalizedClientManifest } from '@sourceog/genbook/types';

import { resolveRouteComponents } from '../rsc/compat-route-loader.js';
import { renderFlight } from '../rsc/compat-renderer.js';
import { RenderError } from '@sourceog/genbook/errors';
import React from 'react';

// ---------------------------------------------------------------------------
// Render Mutex
// ---------------------------------------------------------------------------

/**
 * A simple mutex to ensure that only one render stream writes to the 
 * parentPort at a time. This prevents chunk interleaving where 
 * Route A's chunks mix with Route B's chunks.
 */
let renderMutex: Promise<unknown> = Promise.resolve();

function enqueueRender(task: () => Promise<void>): Promise<void> {
  // Chain the task onto the mutex promise
  const currentChain = renderMutex.then(() => task());
  
  // Update the mutex, catching any errors to avoid unhandled rejection
  renderMutex = currentChain.catch(() => undefined);
  
  return currentChain;
}

// ---------------------------------------------------------------------------
// Request Handler
// ---------------------------------------------------------------------------

/**
 * Handles incoming render requests from the parent thread.
 * 
 * @param message - The raw message object.
 * @param manifest - The loaded client manifest.
 * @param port - The parent port to send responses to.
 */
export function handleRenderRequest(
  message: unknown, 
  manifest: NormalizedClientManifest,
  port: MessagePort
): void {
  const msg = message as WorkerRenderRequest;

  // 1. Validate Message Type
  if (msg.type !== 'render') return;

  // 2. Enqueue Render via Mutex
  void enqueueRender(async () => {
    const { requestId, payload } = msg;
    const { route, context, timeoutMs = 5000 } = payload;

    try {
      // 3. Load Route Components
      const routeComponents = await resolveRouteComponents(route);

      // 4. Construct React Element Tree
      // (Simplified: Full implementation would handle layouts/wrappers)
      const element = React.createElement(
        routeComponents.component as React.ComponentType<any>,
        {
          params: context.params,
          searchParams: Object.fromEntries(context.query || []),
        }
      );

      // 5. Render Flight Stream
      // The renderer handles the stream internally and returns chunks
      const chunks = await renderFlight(element, manifest, {
        routeId: route.id,
        timeoutMs,
      });

      // 6. Send Success Result
      const resultMsg: WorkerMessage = {
        type: 'render_result',
        requestId,
        route: route.id,
        pathname: route.pathname,
        renderContextKey: context.renderContextKey,
        result: {
          format: 'react-flight-text',
          chunks,
          usedClientRefs: [], // Could be extracted if needed
        },
      };
      
      port.postMessage(resultMsg);

    } catch (error) {
      // 7. Handle Errors
      const err = error instanceof Error ? error : new Error(String(error));
      
      const errorMsg: WorkerMessage = {
        type: 'render_error',
        requestId,
        route: route.id,
        pathname: route.pathname,
        renderContextKey: context.renderContextKey,
        error: err.message,
        code: (error as any)?.code || 'RENDER_FAILURE',
        stack: err.stack,
      };

      port.postMessage(errorMsg);
    }
  });
}

// sourceog-renderer/src/workers/worker-types.ts
// Alibaba CTO 2027 Standard — Worker Internal Types

import type { Worker } from 'node:worker_threads';
import type { ActiveRenderRequest } from './worker-router.js'; // Conceptual import

/**
 * Represents a worker instance managed by the pool.
 */
export interface PooledWorker {
  /** The underlying Node.js Worker instance. */
  worker: Worker;

  /** Unique identifier (usually threadId). */
  id: number;

  /** Is the worker currently processing a request? */
  busy: boolean;

  /** Total requests processed by this worker instance. */
  requestCount: number;

  /** Timestamp of the last assignment. */
  lastUsed: number;

  /** Current active request being processed (if any). */
  currentRequest?: ActiveRenderRequest;

  /** Is the worker scheduled for termination? */
  stopping: boolean;
}

/**
 * Configuration for spawning a new worker.
 */
export interface WorkerSpawnOptions {
  /** Path to the worker entry script. */
  entryPath: string;

  /** Path to the manifest file. */
  manifestPath?: string;

  /** Arguments passed to the Node.js process (e.g., --conditions). */
  execArgv?: string[];

  /** Data passed via workerData. */
  workerData?: Record<string, unknown>;
}

/**
 * Health status report for a worker.
 */
export interface WorkerHealthStatus {
  id: number;
  isAlive: boolean;
  isBusy: boolean;
  uptimeMs: number;
  requestCount: number;
  memoryUsage?: NodeJS.MemoryUsage;
}
